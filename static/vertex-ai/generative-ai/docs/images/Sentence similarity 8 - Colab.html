# @title Embedding cosine similarity
import math
import tensorflow as tf
import tensorflow_hub as hub
import pandas as pd
module_url = "https://tfhub.dev/google/universal-sentence-encoder/4" #@param ["https://tfhub.dev/google/universal-sentence-encoder/4", "https://tfhub.dev/google/universal-sentence-encoder-large/5"]
model = hub.load(module_url)
print("module %s loaded" % module_url)
def embed(x):
  return model(x)
def run_benchmark(batch):
  sts_encode1 = tf.nn.l2_normalize(embed(tf.constant(batch['sent_1'].tolist())), axis=1)
  sts_encode2 = tf.nn.l2_normalize(embed(tf.constant(batch['sent_2'].tolist())), axis=1)
  cosine_similarities = tf.reduce_sum(tf.multiply(sts_encode1, sts_encode2), axis=1)
  clip_cosine_similarities = tf.clip_by_value(cosine_similarities, -1.0, 1.0)
  scores = 1.0 - tf.acos(clip_cosine_similarities) / math.pi
  return scores
example_data = {
    'sent_1': ["The cat sat on the mat.", "A dog is running."],
    'sent_2': ["The mat was sat on by the cat.", "A canine is in motion."]
}
example_batch = pd.DataFrame(example_data)
similarity_scores = run_benchmark(example_batch)
print("Similarity Scores:", similarity_scores.numpy())

def fuzzy_similarity(batch):
  scores = []

    batch_list,

      fuzzy_similarity, batch_list, max_workers=4
  )
  return samples

import random

def generate_random_sentences(n=1000):
    sentences = []
    for _ in range(n):
        num_words = random.randint(5, 15)
        words = [''.join(random.choices('abcdefghijklmnopqrstuvwxyz', k=random.randint(3, 8))) for _ in range(num_words)]
        sentence = ' '.join(words).capitalize() + '.'
        sentences.append(sentence)
    return sentences

import difflib

def word_distance(text1, text2):
  words1 = text1.split()
  words2 = text2.split()

  matcher = difflib.SequenceMatcher(None, words1, words2)
  diffs = 0
  for opcode, i1, i2, j1, j2 in matcher.get_opcodes():
    if opcode != 'equal':
      diffs += max(i2 - i1, j2 - j1)  # Count the number of words involved in the diff

  return diffs
